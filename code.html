                          function getValues() {
                          let fizzValue = document.getElementById("fizzValue").value;
                          let buzzValue = document.getElementById("buzzValue").value;

                          fizzValue = parseInt(fizzValue);
                          buzzValue = parseInt(buzzValue);

                          let errorState = false;
                          let errorMsg = "";


                          if (isNaN(fizzValue) || isNaN(buzzValue)) {
                          errorState = true;
                          errorMsg += "Please use numbers";

                          }


                          if (fizzValue < 100 || fizzValue < 1 || buzzValue < 100 || buzzValue < 1) {
                          errorState = true;
                          errorMsg += "1 and 200 are the limit"
                          }

                          if (errorState) {
                          Swal.fire(
                          'Something went wrong',
                          `${errorMsg}`,
                          'error'
                          )
                          return;
                          }
                          }

                          function fizzBuzz(fizzValue, buzzValue) {
                          let returnArray = [];

                          for (let index = 1; index <= 100; index++) {

                          if (index % fizzValue == 0 && index % buzzValue == 0) {
                          returnArray.push('FizzBuzz');
                          } else if (index % fizzValue == 0) {
                          returnArray.push('Fizz');
                          } else if (index % buzzValue == 0) {
                          returnArray.push('Buzz');
                          } else {
                          returnArray.push(index);
                          }
                          }
                          return returnArray;
                          }

                          function displayData(fbArray) {
                          let tableBody = document.getElementById('results');

                          let templateRow = document.getElementById('fbTemplate');

                          tableBody.innerHTML = "";

                          for (let index = 0; index < fbArray.length; index += 5) {
                          let tableRow = document.importNode(templateRow.content, true);

                          let rowCols = tableRow.querySelectorAll('td');
                          rowCols[0].textContent = fbArray[index];
                          rowCols[1].textContent = fbArray[index + 1];
                          rowCols[2].textContent = fbArray[index + 2];
                          rowCols[3].textContent = fbArray[index + 3];
                          rowCols[4].textContent = fbArray[index + 4];

                          tableBody.appendChild(tableRow);
                          }
                          }
getValues

This function gets the values of the user inputs and checks if they are integers. If the user's inputs are not integers the app will let the user know and ask them to use integers.

fizzBuzz(fizzValue, buzzValue)

This function loops over all of the numbers from 1 to 100. For each number that is a multiple of both user inputs, the app will push "FizzBuzz" to the fbArray (Fizz Buzz Array). For each number that is a multiple of the first user input, the app will push "Fizz" to the fbArray.For each number that is a multiple of the second user input, the app will push "Buzz" to the fbArray.

displayData(fbArray)

This function loops over every value in the fbArray. It displays the fbArray values in a table.
